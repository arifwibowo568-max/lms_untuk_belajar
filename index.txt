  <script>
    // ========================================================
    // PASTIKAN URL WEB APP INI SUDAH BENAR
    // ========================================================
    const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbxAI1D3xjf2AACtC4OwTBI7yaEy0CXEsH-TTLkYHT1-UWlpaW7t-FOSHAtGW-xa7m4r/exec"; 

    let IS_AUTH = sessionStorage.getItem('IS_ADMIN') === 'true';
    let DB = { config: {}, content: {} };
    let currentAdminSheet = 'Berita';
    let currentAdminHeaders = [];
    let currentEditId = null;

    document.addEventListener("DOMContentLoaded", function() {
      AOS.init({ once: true, offset: 50 });
      initView();
    });

    // --- FUNGSI FETCH UNTUK REST API (GITHUB) ---
    async function fetchAPI(action, params = "") {
        try {
            const response = await fetch(`${SCRIPT_URL}?action=${action}${params}`);
            return await response.json();
        } catch (error) { throw error; }
    }

    async function postAPI(data) {
        try {
            const response = await fetch(SCRIPT_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                body: JSON.stringify(data)
            });
            return await response.json();
        } catch (error) { throw error; }
    }

    // --- FUNGSI RENDER MENGGUNAKAN IFRAME UNTUK VIDEO DRIVE ---
    // Karena Google Drive memblokir tag HTML5 <video> untuk file di atas beberapa MB,
    // kita WAJIB menggunakan <iframe> agar video tidak blank hitam.
    function renderMedia(url, classes, alt = "Media") {
        if (!url) return '';
        if (url.includes('/preview')) {
            // Pointer-events:none di card agar klik masuk ke detail berita, bukan klik play iframe
            return `<iframe src="${url}" class="${classes} drive-iframe" style="pointer-events:none;" allowfullscreen></iframe>`;
        } else {
            return `<img src="${url}" class="${classes}" alt="${alt}" loading="lazy">`;
        }
    }

    function initView() {
      document.querySelectorAll('.view-section').forEach(v => v.classList.remove('active'));
      if (IS_AUTH) {
        document.getElementById('view-admin').classList.add('active');
        loadAdminDashboard();
      } else {
        document.getElementById('view-site').classList.add('active');
        loadSiteData();
      }
    }

    function hideLoading() {
      const loader = document.getElementById('loading-screen');
      loader.style.opacity = '0';
      setTimeout(() => loader.style.display = 'none', 400);
      document.getElementById('loading-text').innerText = "Memuat Sistem...";
    }

    function showLoading(msg = "Memuat Sistem...") {
      const loader = document.getElementById('loading-screen');
      document.getElementById('loading-text').innerText = msg;
      loader.style.display = 'flex';
      void loader.offsetWidth;
      loader.style.opacity = '1';
    }

    function showLoginView() {
      document.querySelectorAll('.view-section').forEach(v => v.classList.remove('active'));
      document.getElementById('view-login').classList.add('active');
    }

    function showSiteView() {
      document.querySelectorAll('.view-section').forEach(v => v.classList.remove('active'));
      document.getElementById('view-site').classList.add('active');
      if(Object.keys(DB.content).length === 0) loadSiteData();
      else nav('home');
    }

    function nav(section) {
      document.querySelectorAll('.site-section').forEach(s => s.classList.add('hidden'));
      document.getElementById('section-' + section).classList.remove('hidden');
      document.querySelectorAll('.nav-menu a').forEach(a => a.classList.remove('active-link'));
      const activeLink = document.getElementById('link-' + section);
      if(activeLink) activeLink.classList.add('active-link');
      if (section !== 'home' && section !== 'baca') {
        renderSection(section);
      }
      window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    async function loadSiteData() {
      showLoading("Memuat Data Website...");
      try {
        DB.config = await fetchAPI('getSiteConfig');
        renderConfig();
        
        DB.content = await fetchAPI('getContentData');
        renderHome();
        hideLoading();
      } catch (e) {
        hideLoading();
        Swal.fire('Error', 'Gagal terhubung ke server database.', 'error');
      }
    }

    function renderConfig() {
      const aset = DB.config.aset || [];
      const logo = aset.find(r => r[1] === 'Logo');
      if (logo && logo[2]) document.getElementById('logo-img').src = logo[2];
      const banner = aset.find(r => r[1] === 'Banner');
      if (banner && banner[2]) document.getElementById('hero-banner').style.backgroundImage = `url('${banner[2]}')`;

      const landing = DB.config.landing || [];
      const title = landing.find(r => r[1] === 'Hero_Title');
      if (title && title[2]) document.getElementById('hero-title').innerText = title[2];
      const sub = landing.find(r => r[1] === 'Hero_Subtitle');
      if (sub && sub[2]) document.getElementById('hero-subtitle').innerText = sub[2];

      const medsos = DB.config.medsos || [];
      let medsosHtml = '';
      medsos.forEach(m => {
        if(m[1] && m[2]) {
          let icon = m[3] ? m[3] : `fa-${m[1].toLowerCase()}`;
          if(!icon.includes('fa-')) icon = 'fa-' + icon;
          let prefix = (icon.includes('map') || icon.includes('envelope') || icon.includes('phone')) ? 'fas' : 'fab';
          medsosHtml += `<a href="${m[2]}" target="_blank" class="text-slate-400 hover:text-white transition text-2xl" title="${m[1]}"><i class="${prefix} ${icon}"></i></a>`;
        }
      });
      document.getElementById('footer-medsos').innerHTML = medsosHtml;
    }

    function renderHome() {
      const landing = DB.config.landing || [];
      const kepsekRow = landing.find(r => r[1] === 'Kepala_Sekolah');

      if (kepsekRow) {
        if(document.getElementById('kepsek-nama')) document.getElementById('kepsek-nama').innerText = kepsekRow[2] || 'Kepala Madrasah';
        if(document.getElementById('kepsek-sambutan')) document.getElementById('kepsek-sambutan').innerHTML = kepsekRow[3] || '';
        if(document.getElementById('kepsek-img') && kepsekRow[4]) document.getElementById('kepsek-img').src = kepsekRow[4];
      }
      const berita = DB.content.berita || [];
      let html = '';
      if(berita.length === 0) {
        html = '<div class="col-12 text-center text-slate-500 py-10">Belum ada berita dipublikasikan.</div>';
      } else {

    const getBase64 = (file) => new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.readAsDataURL(file);
        reader.onload = () => resolve(reader.result);
        reader.onerror = error => reject(error);
    });

    const compressImage = (file, maxWidth, maxHeight, forceSquare = false) => new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = (event) => {
        const img = new Image();
        img.onload = () => {
          const canvas = document.createElement('canvas');
          let width = img.width;
          let height = img.height;
          let applyForceSquare = forceSquare;
          if(currentAdminSheet === 'AsetGambar' || currentAdminSheet === 'MediaSosial' || currentAdminSheet === 'LandingPage' || currentAdminSheet === 'InfoSPMB') {
            applyForceSquare = false;
          }
          if (applyForceSquare) {
            canvas.width = maxWidth;
            canvas.height = maxHeight;
            const ctx = canvas.getContext('2d');
            const size = Math.min(width, height);
            ctx.drawImage(img, (width - size) / 2, (height - size) / 2, size, size, 0, 0, maxWidth, maxHeight);
          } else {
            if (width > height) {
              if (width > maxWidth) { height = Math.round((height *= maxWidth / width)); width = maxWidth; }
            } else {
              if (height > maxHeight) { width = Math.round((width *= maxHeight / height)); height = maxHeight; }
            }
            canvas.width = width; canvas.height = height;
            const ctx = canvas.getContext('2d');
            ctx.drawImage(img, 0, 0, width, height);
          }
          resolve(canvas.toDataURL(file.type === 'image/png' && !applyForceSquare ? 'image/png' : 'image/jpeg', 0.8));
        };
        img.onerror = error => reject(error);
        img.src = event.target.result;
      };
      reader.onerror = error => reject(error);
    });

    async function uploadMediaInline(file, context, isVideo) {
      showLoading(`Mengunggah ${isVideo ? 'Video' : 'Gambar'} ke Editor...`);
      try {
        let base64Str = isVideo ? await getBase64(file) : await compressImage(file, 800, 800, false);
        const fileName = (isVideo ? "video_" : "inline_") + new Date().getTime() + "_" + file.name;
        
        const res = await postAPI({ action: 'uploadMedia', base64Data: base64Str, fileName: fileName });
        hideLoading();
        
        if(res.success) {
          if(isVideo) {
              const node = document.createElement('div');
              node.innerHTML = `<br><iframe src="${res.url}" width="100%" class="drive-iframe" style="border-radius:12px;" allowfullscreen></iframe><br>`;
              context.invoke('editor.insertNode', node);
          } else {
              context.invoke('editor.insertImage', res.url);
          }
        } else {
          Swal.fire('Gagal Upload', 'Error: ' + res.message, 'error');
        }
      } catch(e) {
        hideLoading(); Swal.fire('Error', 'Gagal memproses file.', 'error');
      }
    }

    async function saveAdminData() {
      let obj = {};
      let isValid = true;

      document.querySelectorAll('.admin-input').forEach(input => {
        if(input.classList.contains('rich-text')) {
          obj[input.dataset.header] = $(input).summernote('code');
        } else {
          obj[input.dataset.header] = input.value;
          if(input.type !== 'hidden' && !input.value) {
            isValid = false;
            input.classList.add('is-invalid');
          } else {
            input.classList.remove('is-invalid');
          }
        }
      });

      if(!isValid) { Swal.fire('Peringatan', 'Harap lengkapi semua data wajib!', 'warning'); return; }
      
      const fileInputs = document.querySelectorAll('.admin-file');
      for (let fileInput of fileInputs) {
        const header = fileInput.dataset.header;
        if (fileInput.files && fileInput.files[0]) {
            const file = fileInput.files[0];
            if (file.size > 15 * 1024 * 1024) { Swal.fire('Peringatan', 'Ukuran file media maksimal 15MB', 'warning'); return; }
          
            showLoading("Mengunggah Media...");
            try {
                let base64Str;
                if (file.type.startsWith('video/')) {
                    base64Str = await getBase64(file);
                } else {
                    base64Str = await compressImage(file, 500, 500, true);
                }
                const fileName = "media_" + new Date().getTime() + "_" + file.name;
                const uploadRes = await postAPI({ action: 'uploadMedia', base64Data: base64Str, fileName: fileName });

                if (uploadRes.success) {
                    obj[header] = uploadRes.url;
                } else {
                    Swal.fire('Gagal Upload', 'Detail: ' + uploadRes.message, 'error');
                    hideLoading(); return;
                }
            } catch (e) {
                Swal.fire('Error', 'Terjadi kesalahan saat upload', 'error');
                hideLoading(); return;
            }
        }
      }

      showLoading("Memperbarui Database...");
      try {
        let res;
        if (currentEditId) {
          res = await postAPI({ action: 'updateData', sheetName: currentAdminSheet, id: currentEditId, dataObj: obj });
        } else {
          res = await postAPI({ action: 'insertData', sheetName: currentAdminSheet, dataObj: obj });
        }
        
        hideLoading();
        if(res.success) {
          bootstrap.Modal.getInstance(document.getElementById('adminModal')).hide();
          Swal.fire('Sukses', res.message, 'success');
          loadAdminModule(currentAdminSheet);
          DB.content = {};
        } else {
          Swal.fire('Error', res.message, 'error');
        }
      } catch (e) {
        hideLoading(); Swal.fire('Error', 'Gagal menyimpan data.', 'error');
      }
    }

    function deleteAdminData(id) {
      Swal.fire({
        title: 'Yakin hapus data?', text: "Data tidak dapat dikembalikan!", icon: 'warning', showCancelButton: true, confirmButtonColor: '#d33', cancelButtonColor: '#3085d6', confirmButtonText: 'Ya, Hapus!'
      }).then(async (result) => {
        if (result.isConfirmed) {
          showLoading("Menghapus Data...");
          try {
            const res = await postAPI({ action: 'deleteData', sheetName: currentAdminSheet, id: id });
            hideLoading();
            if(res.success) { Swal.fire('Terhapus!', res.message, 'success'); loadAdminModule(currentAdminSheet); DB.content = {}; }
            else { Swal.fire('Error', res.message, 'error'); }
          } catch(e) {
            hideLoading(); Swal.fire('Error', 'Gagal menghapus data.', 'error');
          }
        }
      });
    }
